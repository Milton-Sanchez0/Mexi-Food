<script>
function toggleIngredientes() {
let platillo = document.getElementById("platillo").value
let ingredientesDiv = document.getElementById("ingredientes")
if (platillo === "Torta de Pollo") {
ingredientesDiv.style.display = "block"
} else {
ingredientesDiv.style.display = "none"
}
}
function toggleAguaOpciones() {
let bebida = document.getElementById("bebida").value
let aguaOpcionesDiv = document.getElementById("aguaOpciones")
if (bebida === "Agua Natural") {
aguaOpcionesDiv.style.display = "block"
} else {
aguaOpcionesDiv.style.display = "none"
}
}
function obtenerUbicacion() {
if (navigator.geolocation) {
navigator.geolocation.getCurrentPosition(
function (position) {
let lat = position.coords.latitude
let lon = position.coords.longitude
document.getElementById("ubicacion").value =
`Lat: ${lat}, Lon: ${lon}`
},
function (error) {
alert("No se pudo obtener tu ubicación.")
},
)
} else {
alert("La geolocalización no está soportada por tu navegador.")
}
}
function enviarPedido() {
let nombre = document.getElementById("nombre").value
let platillo = document.getElementById("platillo").value
let bebida = document.getElementById("bebida").value
let ingredientes = []
document
.querySelectorAll("#ingredientes input:checked")
.forEach((el) => ingredientes.push(el.value))
let ingredientesTexto =
ingredientes.length > 0 ? ` con ${ingredientes.join(", ")}` : ""
let aguaSeleccion = ""
if (bebida === "Agua Natural") {
let aguaOpciones = document.querySelector(
'input[name="agua"]:checked',
)
if (aguaOpciones) {
aguaSeleccion = ` de ${aguaOpciones.value}`
}
}
let ubicacion = document.getElementById("ubicacion").value
let telefono = "8445067926" // 8445067926
let mensaje = `Hola, mi nombre es ${nombre}. Quiero pedir ${platillo}${ingredientesTexto} con
${bebida}${aguaSeleccion}. Mi ubicación es: ${ubicacion}.`
let url = `https://wa.me/${telefono}?text=${encodeURIComponent(mensaje)}`
// Muestra el mensaje de agradecimiento
document.getElementById("pedidoForm").style.display = "none" // Oculta el formulario
document.getElementById("graciasMensaje").style.display = "block" // Muestra el mensaje de
agradecimiento
// Redirige a WhatsApp después de 3 segundos
setTimeout(function () {
window.open(url, "_blank")
}, 3000) // 3 segundos
}
</script>
